\section{Trails}

\begin{question} \label{Q8effrfrf}
The differentialble function
\[
      h = f(s) \, , \, 0\leq s \leq 3 ,
\]
expresses the altitude (in meters) of a trail in terms of the distance (in km) from the trailhead.

The differentiable function
\[
    s = g(t) \, , \, 0\leq t \leq 4,
\]
expresses your distance from the trailhead (in km) as you hike along the trail in terms of the number of hours past noon.

\begin{enumerate}
\item Suppose three kilometers from the trailhead the scaling factor that converts a small change in the distance from the trailhead (in km) into an accurate approximation of the corresponding change in altitude (in meters) is $-250$. Suppose also that $f(3)=1200$.

\begin{enumerate}
\item What are the units of the scaling factor.

\item Explain the meaning of the scaling factor in terms of \emph{specific} small changes.

\item Approximate the change
\[
 \Delta h = h - 1200
\]
in altitude in terms of a small change
\[
  \Delta s = s -3
\]
in distance from the trailhead.

\item Use the Leibniz evaluation notation to write the scaling factor as a derivative.
\end{enumerate}

\item Suppose also that $g(1.5) = 3$ and that
\[
    \frac{ds}{dt} \Big|_{t=1.5} = 2 .
\]

\begin{enumerate}
\item What are the units of this derivative?

\item Use everyday Engish to explain the meaning of the derivative.

\item Approximate the change
\[
  \Delta s = s - 3
\]
in your distance from the trailhead (in km) in terms of a small change
\[
     \Delta t = t - 1.5
\]
in the number of hours past noon.
\end{enumerate}

\item Use the results of parts (a) and (b) to approximate your change in altitude ($\Delta h$) in terms of a small change in the number of hours past noon ($\Delta t$).

\item Use everyday English to explain the meaning of the scaling factor in part (c).

\item Our conclusion is 
\[
    \frac{dh}{dt}\Big|_{t=1.5} = \frac{dh}{\answer{ds}}\Big|_{s=\answer{3}}    \cdot \frac{\answer{ds}}{\answer{dt}}\Big|_{t=\answer{1.5}}
\]
or more generally that the derivative of the function
\[
    h = f(g(t)) \, , \, 0\leq t \leq 4,
\]
expressing  your altitude (in km) in terms of the number of hours past noon is
\[
  \frac{dh}{dt}\Big|_{t=a} = \frac{dh}{\answer{ds}}\Big|_{s=\answer{f(a)}}    \cdot \frac{\answer{ds}}{\answer{dt}}\Big|_{t=\answer{a}} .
\]
\end{enumerate}

\end{question}


\begin{question} \label{WQPlkfd3}

\[
      h = f(s)  = 2100 - 200 \sin s   \, , \, -1\leq s \leq 3 ,
\]
expresses the altitude (in meters) of a $4$-km trail in terms of the signed distance (in km) from the one-kilometer mark. 

The function
\[
    s = g(t)  =  - 1 + 0.35 t + \frac{t^{2}}{5} \, , \, 0\leq t \leq 4,
\]
expresses your signed distance from the one-kilometer mark as you hike along the trail in terms of the number of hours past noon.

\begin{enumerate}
\item Find your speed at 12:00pm. Include units. Use the Leibniz evaluation notation.

\item Find the steepness of the trail beneath your feet at 12:00pm. Include units. Use the Leibniz evaluation notation. Explain the meaning of the steepness as a scaling factor.

\item Use the Leibniz evaluation notation to express your rate of ascent at 12:00pm in terms of the derivatives in parts (a) and (b). Then compute this rate of ascent. Include units.

\item Find the angle the trai beneath your feet makes with the horizontal at 12:00pm.

\end{enumerate}
\end{question}

\section{Purchasing Power}
\begin{question} \label{Qpdf3r3r0}
\begin{enumerate}

item Find a function
\[
    N = w(p) \, , \, p>0 ,
\]
that expresses the number of barells of oil you can buy with $\$10,000$ in terms of the price (in dollars/barell).

\item The function
\[
        P = h(t)  = 100-10\sin t \, , \, -5 \leq 5 ,
\]
expresses the price of oil (in dollars/barrel) in terms of the number of hours past noon.

\begin{enumerate}

\item Is the price of oil increasing or decreasing at noon? At what rate? Use the Leibniz evaluation notation for the derivative.

\item Evaluate the derivative 
\[
  \frac{dN}{dp}
\]
at the price of oil at noon. Use the Leibniz notation. Interpret its meaning as a scaling factor.

\item Is the number of barrels you can buy with $\$10,000$ increasing or decreasing at noon? At what rate? Use the Leibniz evaluation notation. 
\end{enumerate}

\end{enumerate}
\end{question}

\begin{enumerate}

\[
     h = f(\theta) \, , \, -\pi \leq \theta \leq \pi ,
\]
that expresses your height above the ground (in feeet) in terms of the wheel's signed angle of rotation, measured in radians from the downward-pointing vertical. 

\item Find an expression for the derivative $dh/d\theta$. Include units. Interpret its meaning as a scaling factor.
\end{enumerate}
