\begin{enumerate}

\begin{freeResponse}
\end{freeResponse}

\item Drag the slider $u$ (another name for $\theta$) in Line 2 to guess the value of the limit
\[
   \lim_{\theta\to 0} \frac{\sin (2\theta)}{\theta}.
\]
\begin{freeResponse}
\end{freeResponse}

\item To compute the limit we'll make the substitution $u = 2\theta$. Then
\[
   \theta = \answer{u/2}
\]
and as $\theta\to 0$, 
\[
  u = 2\theta \to \answer{0}.
\]
So
\begin{align*}
\lim_{\theta\to 0} \frac{\sin (2\theta)}{\theta} &= \lim_{u\to \answer{0}} \frac{\answer{\sin u}}{\answer{u/2}} \\
                               &= \answer{2} \lim_{u \to 0} \frac{\answer{\sin u}}{\answer{u}} \\
                               &= 2 (\answer{1}) \\
                               &= \answer{2} .
\end{align*}

\end{enumerate}
\end{example}

\section{Sine Proof 1}

\begin{example}  \label{ExMV21er}
Here's one way to show
\[
   \frac{d}{d\theta}\left(  \sin\theta \right) = \cos\theta.
\]
 It uses the identity
\[
 \sin(\alpha + \beta) = \sin\alpha \cos\beta + \cos \alpha \sin\beta.
\]


\begin{enumerate}

\item First use the result of Example 2 to evaluate the derivative
\[
   \frac{d}{d\theta}\left(  \sin\theta \right) \Big|_{\theta=0} .
\]

\item Use the graph of the function $y=\cos\theta$ to evaluate the derivative
\[
  \frac{d}{d\theta}\left(  \cos\theta \right) \Big|_{\theta=0} .
\]
Explain your reasoning with a picture.
\begin{freeResponse}
\end{freeResponse}


\item Explain why
\[
      \frac{d}{d\theta}\left(  \sin\theta \right) \Big|_{\theta = \theta_0} = \frac{d}{d\theta}\left(  \sin(\theta + \theta_0)\right) \Big|_{\theta = 0}.
\]
Do this by graphing the two functions.
\begin{freeResponse}
\end{freeResponse}

\item Then use parts (a)-(c).
\begin{align*}
 \frac{d}{d\theta}\left(  \sin\theta \right) \Big|_{\theta = \theta_0} &= \frac{d}{d\theta}\left(  \sin(\theta + \theta_0)\right) \Big|_{\theta = 0} \\
                    &= \frac{d}{d\theta} \left( \sin\theta\cos\theta_0 + \cos\theta \sin\theta_0  \right)\Big|_{\theta = \theta_0} \\
                    &= \frac{d}{d\theta} \left( \sin\theta\cos\theta_0 \right)\Big|_{\theta = \theta_0} + \frac{d}{d\theta} \left( \cos\theta\sin\theta_0 \right)\Big|_{\theta = \theta_0}  \\
                    &= \left( \cos\theta_0 \right)  \frac{d}{d\theta} \left( \sin\theta \right)\Big|_{\theta = \theta_0} + (\sin\theta_0) \frac{d}{d\theta}\left( \cos\theta \right)\Big|_{\theta = \theta_0} \\
                    &= (\cos\theta_0)(\answer{1}) + (\sin\theta_0)(\answer{0})  \\
                    & = \answer{\cos} \theta_0 .
\end{align*}

\end{enumerate}
\end{example}
